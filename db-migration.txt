#UPDATE ENV VARIABLE DB_MIGRATE IN CORRESPONDING DOCKER COMPOSE FILE
DB_MIGRATE=True 
#START UP CONTAINER
#GET INSIDE RUNNING CONTAINER
docker exec -it chatbot bash 

#IF NO MIGRATIONS FOLDER 
flask db init --directory=[DIRECTORY NAME]

#IF FIRST TIME 
#indicate that the current state of the database represents the application of all migration
flask db stamp head (--directory=[DIRECTORY NAME]) 

#FOR DEV
flask db migrate -m "..."
flask db upgrade 


#FOR PROD
flask db migrate -m "..." --directory="migrations_prod"
flask db upgrade --directory="migrations_prod"


#BRING THE MIGRATIONS FOLDER BACK IN LOCAL MACHINE FROM CONTAINER
docker cp {container ID}:/app/{migrations_folder} .
